    mainTest.cpp
    ./testsuits/example_failure.cpp
    ./testsuits/example_success.cpp
)

    gtest
    gmock
)

    NAME ${PROJECT_NAME}_tests
    COMMAND ${PROJECT_NAME}_tests
)
